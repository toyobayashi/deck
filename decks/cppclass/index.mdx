# 怎么写算是用户声明？

---

<CodeSurfer>

```cpp subtitle="1. 声明且定义"
struct X {
  X() { /* ... */ }  // 声明并 inline 定义
};
```

```cpp subtitle="2. 只声明，不定义，或在类外定义"
struct X {
  X() { /* ... */ }  // 声明并 inline 定义
  X();  // 只声明，不定义
};
```

```cpp subtitle="3. 让编译器生成默认定义"
struct X {
  X() { /* ... */ }  // 声明并 inline 定义
  X();  // 只声明，不定义
  X() = default;  // 让编译器生成默认定义
};
```

```cpp subtitle="4. 阻止编译器生成（该成员函数存在，但不可用）"
struct X {
  X() { /* ... */ }  // 声明并 inline 定义
  X();  // 只声明，不定义
  X() = default;  // 让编译器生成默认定义
  X() = delete;  // 阻止编译器生成（该成员函数存在，但不可用）
};
```

```diff subtitle="以上是默认构造函数的用户声明例子，6 个特殊成员都可以这样写"
```

# 未声明和删除有什么区别

---

<CodeSurfer>

```cpp
struct X {

};

int main() {
  return 0;
}
```

```cpp
struct X {
  template <typename... Args>
  X(Args&&... args);

  // 当用户声明了任意构造函数
  // 默认构造函数是 未声明 的
  // 但 X 可以通过变参构造函数
  // 被无参构造
};

int main() {
  // 调用 X::X<>()
  X x;
  return 0;
}
```

```cpp
struct X {
  template <typename... Args>
  X(Args&&... args);

  X() = default;

  // 默认构造函数可用
};

int main() {
  // 重载决议选择 X::X()
  X x;
  return 0;
}
```

```cpp
struct X {
  template <typename... Args>
  X(Args&&... args);

  X() = delete;

  // 默认构造函数存在，但不可用
};

int main() {
  // 重载决议选择 X::X()
  // 编译不通过
  X x;
  return 0;
}
```

```diff subtitle="未声明的特殊成员 不参与 重载决议，删除的特殊成员 会参与 重载决议"
```

# 移动到底怎么移

---

<CodeSurfer>

```cpp title="编译器默认生成的移动构造函数做了什么"
class Derived : public Base {
 private:
  Member m_;
 public:
  Derived(Derived&& d):
    Base(static_cast<Base&&>(d)),
    m_(static_cast<Member&&>(d.m_)) {}
};
```

```cpp title="用户定义的移动构造函数一般要做什么"
class Derived : public Base {
 private:
  Member m_;
 public:
  Derived(Derived&& d):
    Base(std::move(d)),
    m_(std::move(d.m_))
  {
    d.SetToResourcelessState();
  }
};
```

```cpp title="编译器默认生成的移动赋值运算符做了什么"
class Derived : public Base {
 private:
  Member m_;
 public:
  Derived& operator=(Derived&& d) {
    Base::operator=(static_cast<Base&&>(d));
    m_ = static_cast<Member&&>(d.m_);
    return *this;
  }
};
```

```cpp title="用户定义的移动赋值运算符一般要做什么"
class Derived : public Base {
 private:
  Member m_;

  Derived& operator=(Derived&& d) {
    Base::operator=(std::move(d));
    m_ = std::move(d.m_);
    d.SetToResourcelessState();
    return *this;
  }
```

# 如何写好一个类

---

<CodeSurfer>

```cpp subtitle="1. 不写任何特殊成员，让编译器隐式生成"
class X {
 private:
  std::vector<int> v_;
 public:
  // 请双手离开键盘
};
```

```cpp subtitle="2. 写了就 6 个都全部写全"
class X {
 private:
  char* s_;
 public:
  X();
  ~X();
  X(const X&);
  X& operator=(const X&);
  X(X&&);
  X& operator=(X&&);
};
```

```cpp subtitle="3. 只可移动不可拷贝"
class X {
 private:
  FILE* f_;
 public:
  X();
  ~X();
  // X(const X&) = delete;
  // X& operator=(const X&) = delete;
  X(X&&);
  X& operator=(X&&);
};
```

```cpp subtitle="4. 不可拷贝不可移动"
class X {
 private:
  std::mutex m_;
 public:
  X(const X&) = delete;
  X& operator=(const X&) = delete;
};
```

```cpp subtitle="5. 只可拷贝不可移动，不要删除移动成员"
class X {
 public:
  X(const X&);
  X& operator=(const X&);

  // 错误的做法
  X(X&&) = delete;
  X& operator=(X&&) = delete;
};

X GetX();

int main() {
  // 移动构造函数已声明（存在）
  // 不会调用拷贝构造函数
  // 编译不通过
  X x = GetX();

  return 0;
}
```

```cpp 3:4,10:12 subtitle="5. 只可拷贝不可移动，正确的做法"
class X {
 public:
  X(const X&);
  X& operator=(const X&);
};

X GetX();

int main() {
  // 移动构造函数未声明
  // 调用拷贝构造函数
  X x = GetX();

  return 0;
}
```

```cpp subtitle="6. 可继承的基类必须有 虚析构函数"
class Base {
 public:
  virtual ~Base();
  Base(const Base&);
  Base& operator=(const Base&);
  Base(Base&&);
  Base& operator=(Base&&);
};

class Derived : public Base {};

int main() {
  // 基类指针指向派生类
  Base* p = new Derived();
  // 基类析构函数是 virtual 的
  // 派生类的析构函数也会被调用
  // 防止内存泄漏
  delete p;
  // 调用 Derived::~Derived()
  // 调用 Base::~Base()
  {
    // 智能指针也一样
    std::unique_ptr<Base> sp =
      std::make_unique<Derived>();
  }

  return 0;
}
```
